\renewcommand{\thetheoremcounter}{\thesection.\arabic{theoremcounter}}
\newenvironment{template}[2]{
	\refstepcounter{theoremcounter}

  	\textbf {
	    \thetheoremcounter \quad #1  \ifthenelse{\equal{#2}{}}{}{({#2})}
  	}
	\begin{quote}
  	\begingroup\itshape
} {
	\endgroup
	\end{quote}
	\smallskip
}
