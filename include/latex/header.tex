%------------------------- Theorem and Proof Environments -------------------------------------------------

% This section defines all the environments you might use.  Just type
% \begin{theorem, or corollary, or whatever}, then the optional name of the
% theorem inside {} (or empty {} if no name), then body of the theorem,
% corollary, whatever, also inside {} then \end{theorem, corollary, whatever}
%
% Notice when I use them in the paper, I put an optional "argument" to the function
% and this gives a name to the theorem


    \providecommand{\myDefCounter}{section}
    \ifthenelse{ % if
      \equal{\myDefCounter}{section}
    }{ % then
    }{ % else
      \def\themyDefCounter{\thechapter.\arabic{myDefCounter}}
    }

\newcounter{TheoremCounter}
\numberwithin{TheoremCounter}{section}
\newenvironment{theorem}[1][]{
  \refstepcounter{TheoremCounter}
  \vspace{.9cm}
  \noindent    
  \textbf{
    Theorem \theTheoremCounter \quad \ifthenelse{\equal{#1}{}}{}{({#1})}
    \newline
  }
} {
  \par\bigskip  
} 

\newenvironment{lemma}[1]{
  \vspace{.9cm}
  \noindent    
  {\bf Lemma {#1}}
} {
  \vspace{.1cm}
}
\newenvironment{corollary}[1]{
  \vspace{.9cm}
  \noindent
  {\bf Corollary {#1}}
} {
  \vspace{.1cm}
}
\newenvironment{definition}[1]{
  \vspace{.9cm} 
  \noindent 
  {\bf Definition}
} { 
  \vspace{.1cm}
}
\def\qed{\hfill $\Box$}
\renewenvironment{proof} {
  \vspace{.5cm}   
  \noindent
  {\bf Proof: }
} {
  \qed \vspace{1cm}
}
